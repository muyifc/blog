        txt = ""
        isTr = False
        isTd = False
        tdCount = 0
        def handle_starttag(self,tag,attr):
            if tag == 'tr':
                self.isTr = True
                self.tdCount = 0
                self.first = True
            if tag == "td":
                self.isTd = True
                self.tdCount += 1
                # print("start",tag,attr)
        def handle_endtag(self,tag):
            if tag == "tr":
                self.isTr = False
                self.txt += '},\n'
            if tag == "td":
                self.isTd = False
            # print("end",tag)
        def handle_data(self,data):
            # print(data)
            if self.isTr and self.isTd:
                if self.tdCount == 1 or self.tdCount == 2 or self.tdCount == 4:
                    if self.first:
                        self.txt += "{sign=\"" + data.lower() + "\""
                        self.first = False
                    else:
                        if self.tdCount == 2:
                            self.txt += ",name=\""
                        elif self.tdCount == 4:
                            self.txt += "\",code="
                        self.txt += data
