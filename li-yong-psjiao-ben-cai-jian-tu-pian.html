最终代码：

var doc = app.activeDocument;
var originalStem = app.activeDocument.activeLayer;

cutbounds();

// 裁切 基于透明像素
function trim(){
    doc.trim(TrimType.TRANSPARENT,true,true,true,true);
}

// 裁剪4个角
function cutbounds(){
    var width = app.activeDocument.width;
    var height = app.activeDocument.height;
    var maxSide = width / 2;
    if(maxSide > height/2){
        maxSide = height/2;
    }
    // 可交互输入弹框
    var side = prompt("裁剪矩形边长:" + originalStem.name,16);
    while(isNaN(side) || side <= 0 || side >= maxSide){
        // 弹出文本提示框
        alert("请输入有效的边长");
        return;
    }

    trim();

    var width = app.activeDocument.width;
    var height = app.activeDocument.height;

    // 创建图层组
    var stemGroup = app.activeDocument.layerSets.add();
    stemGroup.name = "corners"

    // 四个角的矩形
    var regions = Array(
        Array(Array(0,side),Array(0,0),Array(side,0),Array(side,side)),
        Array(Array(width-side,side),Array(width-side,0),Array(width,0),Array(width,side)),
        Array(Array(width-side,height),Array(width-side,height-side),Array(width,height-side),Array(width,height)),
        Array(Array(0,height),Array(0,height-side),Array(side,height-side),Array(side,height)),
        );
    for(var i=0;i < 4;i++){
        app.activeDocument.activeLayer = originalStem;
        // 选择一个选区
        var selectRect = doc.selection.select(regions[i]);
        // 将选区中的图层复制下来
        doc.selection.copy();
        var newStem = doc.paste();
        newStem.name = "corner_" + i;
        // 将新图层放入图层组中
        newStem.move(stemGroup,ElementPlacement.PLACEATEND);
        var deltaX = 0;
        var deltaY = 0;
        if(regions[i][0][0] != 0){
            deltaX = -(width - side*2);
        }
        if (regions[i][0][1] != side){
            deltaY = -(height - side*2);
        }
        // 移动图层
        newStem.translate(deltaX,deltaY);
    }
    // 隐藏原始图层
    originalStem.visible = false;

    trim();
}
